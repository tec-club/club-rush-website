function sqrt( num ) {
    if ( num < 0 ) {
        return Number.NaN;
    } else {
        var acc = 1e-15;
        var sq = num;
        
        while ( Math.abs( num/sq - sq ) > acc * sq ) {
            sq = (num / sq + sq) / 2.0;
        }
        return sq;
    }
}
